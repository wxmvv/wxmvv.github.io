export NVM_DIR="$HOME/.nvm"
export PATH=$HOME/.nvm/versions/node/v18.14.0/bin/:$PATH
nvm() { . "$NVM_DIR/nvm.sh" ; nvm $@ ; }

注释掉之前添加的nvm启动命令

然后将上面代码中的版本号改为自己想默认启动的node，存入配置文件
